.. py:function:: match_signals(names: list[str] | set[str], *, include: list[str] | None = None, exclude: list[str] | None = None, regex: bool = False) -> list[str]

   Return signal names matching the include/exclude filters.

   :param names: All available signal names.
   :param include: Patterns to include.  If ``None``, all names are included.
   :param exclude: Patterns to exclude.  Applied after include filtering.
   :param regex: If ``True``, patterns are Python regexes.
                 If ``False`` (default), patterns are Unix glob patterns.

   :returns: Matching signal names, in the order they appear in *names*.
   :rtype: list[str]


.. py:function:: select_signals(traces: dict[str, list], *, include: list[str] | None = None, exclude: list[str] | None = None, regex: bool = False) -> dict[str, list]

   Filter a traces dict to only include matching signals.

   :param traces: Signal name → trace list mapping (from ``CompiledModule.run()``).
   :param include: Glob or regex patterns for signals to keep.
   :param exclude: Glob or regex patterns for signals to remove.
   :param regex: Interpret patterns as regex (default: glob).

   :returns: Filtered traces dict.
   :rtype: dict[str, list]


.. py:function:: traces_to_vcd(traces: dict[str, list[tuple[datetime.datetime, int]]], *, module: dau_sim.ir.module.Module | None = None, timescale: str = '1ns', scope: str | None = None) -> str

   Convert simulation traces to a VCD string.

   :param traces: Signal traces from ``CompiledModule.run()``.
   :type traces: dict
   :param module: IR module (used for signal widths and port directions).
                  If None, all signals are assumed 1-bit wire.
   :type module: Module | None
   :param timescale: VCD timescale (e.g. "1ns", "1ps", "10ns").
   :type timescale: str
   :param scope: Top-level scope name. Defaults to module name or "top".
   :type scope: str | None

   :returns: Complete VCD file content.
   :rtype: str
